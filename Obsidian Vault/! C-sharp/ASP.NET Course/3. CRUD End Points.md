### POST Endpoint
```C#
//Program.cs

//this is a post request using 
app.MapPost("games", (CreateGameDto newGame) =>
{
    GameDto game = new(
        games.Count + 1,
        newGame.Name,
        newGame.Genre,
        newGame.Price,
        newGame.ReleaseDate);

    games.Add(game);
    
    return Results.CreatedAtRoute(GetGameEndpoint, new { id = game.id}, game);
});
```
