class Solution {
public:
    vector<int> leaders(vector<int>& nums) {
        if(nums.size() < 2){
            return nums;
        }

        int i = nums.size() - 1;
        vector<int> solution;

        int max = INT_MIN;

        while(i >= 0){
            if(nums[i] > max){
                max = nums[i];
            }
            if(solution.empty() || solution.back() != max){
                solution.push_back(max);
            }
            i--;
        }
        reverse(solution.begin(), solution.end());
        return solution;
    }
};
