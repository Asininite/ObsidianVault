class Solution {
public:
    int primeUptoN(int n) {
        int count = 0;
        for(int i = 1; i <=n; i++){
            int x = checkprime(i);
            count += x
        }
        return count;
    }
    
    int checkprime(int num){
        int limit = static_cast<int>(sqrt(num));
        if(num == 1) return 0;
        if(num == 2) return 1;
        if(num == 3) return 1;
        if(num%2 == 0) return 0;
        if(num%3 == 0) return 0;
        for(int i = 5; i <=limit; i+=6){
            if(num%i == 0 || num%(i+2) == 0){
                return 0;
            }
        }
        return 1;
    }
};
