### O(N) Brute Force
```cpp
class Solution {
public:
    int singleNonDuplicate(vector<int> &nums) {
        int sum = 0;
        for(int i = 0; i < nums.size(); i++){
            sum = sum ^ nums[i];
            // what XOR operation does is simple, it cancels out any duplicates
        }
        return sum;
    }
};
```
