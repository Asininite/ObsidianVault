
### Brute Force (O(N))
```cpp
class Solution{
public:
    int lowerBound(vector<int> &nums, int x){
        int i = 0;
        while (i < nums.size()){
            if(nums[i] >= x){
                return i;
            } else {
                i++;
            }
        }
        return nums.size();
    }
};
```
