
### Brute Force (O(N))
```cpp
class Solution{
public:
    int lowerBound(vector<int> &nums, int x){
        int i = 0;
        while (i < nums.size()){
            if(nums[i] >= x){
                return i;
            } else {
                i++;
            }
        }
        return nums.size();
    }
};
```

### Optimal (O(log N))
```cpp
class Solution{
public:
    int lowerBound(vector<int> &nums, int x){
        int n = nums.size();
        int low = 0;
        int high = n - 1;
        int max = n;

        while(low <= high){
            int mid = (low + high)/2;
            if(nums[mid] >= x){
                max = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return max;
    }
};
```
