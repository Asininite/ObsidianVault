#include <bits/stdc++.h>
using namespace std;

class Solution{
	public:
		vector<int> preorder(TreeNode* root){
			TreeNode* node = root;
			stack<TreeNode*> st;
			vector<int> arr;
			
			while(true){
				if(node != nullptr){
					st.push(node);
					arr.push_back(node->data);
					node = node->left;
				} else {
					if(st.empty()){
						break;
					}
					node = st.top();
					st.pop();
					node = node -> right;
				}
			}
			return arr;
		}
}; // after class definition this is important
